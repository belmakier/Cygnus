192  // For electric transitions:
193  // C(E,Lambda) = 1.116547 * pow(13.889122,Lambda) *
194  // (Lambda - 1)! / (2*Lambda + 1)!!
195  // For magnetic transitions:
196  // C(M,Lambda) = (v/c)C(E,Lambda)/95.0981942:

204  // holders for the states and substates. Note that in the excitation
205  // process, substates are treated individually, with states only ho-
206  // lding information common to all substates.

213  tmpState.SetEta(fReaction.EtaCalc(fNucleus.GetLevelEnergies().at(s)));
214  tmpState.SetPsi(fReaction.GetLabEnergy() - (1 + fA1/fA2) * fNucleus.GetLevelEnergies().at(s));
215  int phase = 1;
216  if(s != 0)
217  phase = TMath::Power(-1, 1 - tmpState.GetJ() - fStates.at(0).GetJ());
218  tmpState.SetPhase(phase);
219  fStates.push_back(tmpState);
220  for(double ss = -fNucleus.GetLevelJ().at(s); ss <= fNucleus.GetLevelJ().at(s); ss++){
221  if(ss == -0)
222  ss = 0;

224  fSubstates.push_back(tmpSubstate);
225  }
226  }

229  // appropriate integration range later

231

233  // the FINAL state, with the connections into that substate from
234  // other states being the important variables

236

238  int s1 = fSubstates.at(ss1).GetStateIndex();
239  for(size_t ss2 = 0; ss2 < fSubstates.size(); ss2++){ // Substate 2 is the INITIAL state
240  // Select the state corresponding to the substate ss1
241  int s2 = fSubstates.at(ss2).GetStateIndex();
242 
243  // Phase convention for matrix elements:
244  int mePhase = TMath::Power(-1,(fStates.at(s2).GetJ()-fStates.at(s1).GetJ()));
245  if(s2 > s1)
246  mePhase = 1;
247

249  double psi2 = fStates.at(s2).GetPsi();
250  // Phase factor for coupling parameter, depends on INITIAL (sub)state only
251  double phase = TMath::Power(-1,fStates.at(s2).GetJ()-fSubstates.at(ss2).GetM());
252 
253  // Difference in wavenumber between final and initial sttae
254  double tmpXi = fStates.at(fSubstates.at(ss1).GetStateIndex()).GetEta() - fStates.at(fSubstates.at(ss2).GetStateIndex()).GetEta();
255 
256  // Holder class for connection information, will be filled with all conne-
257  // ctions between substates ss1 and ss2 and then held in Substate ss1
258  //
259  // See class Connection for details on connection holder class
260  Connection tmpConnection;
261  tmpConnection.SetConnectedState(ss2);
262  // For convenience, we allow all possible Lamda values (E1-6, M1)
263  tmpConnection.SetMaxLambda(fNucleus.GetMaxLambda());
264 
265  // Does this connection actually contain anything?
266  bool connectionFlag = false;
267  for(int l=0;l<fNucleus.GetMaxLambda();l++){ // Loop over multipolarities
268 
269  // If there is no matrix element between these states at this
270  // multipolarity, then we don't need to go any further
271  if(fNucleus.GetMatrixElements().at(l)[s1][s2] == 0)
272  continue;
273 
274  // Because of the indexing convention for magnetic transitions
275  // (M1 = 7), we need to adjust our actual lambda accordingly
276  int lambda = l+1;
277  if(l > 5)
278  lambda = l-5;
279 
280  // If the substate -> substate transition is forbidden on
281  // angular momentum grounds, we can skip this
282  if(TMath::Abs(fSubstates.at(ss1).GetM() - fSubstates.at(ss2).GetM()) > lambda)
283  continue;
284  if(TMath::Abs(fStates.at(fSubstates.at(ss1).GetStateIndex()).GetJ() - fStates.at(fSubstates.at(ss2).GetStateIndex()).GetJ()) > lambda)
285  continue;
286 
287  // Update our maximum wavenumber difference for use in the
288  // integration step later
289  if(tmpXi > XiMax)
290  XiMax = tmpXi;
291 
292  double Power = (2*((double)lambda)-1)/4.;
293 
294  double s = TMath::Power(AAZZ,lambda);
295  double C;
296  if(l > 5)
297  C = CPsi_Mag / s;
298  else
299  C = CPsi_Elec[l] / s;
300 
301  double tmpPsi = C * fZ1 * TMath::Sqrt(fA1) * TMath::Power( psi1 * psi2,Power);
302 
303  // ThreeJ symbol uses 2* values (to allow for half-integer
304  // spin) so we create a few new variables
305  int JInit = 2 * fStates.at(fSubstates.at(ss2).GetStateIndex()).GetJ();
306  int L2 = 2 * lambda;
307  int JFinal = 2 * fStates.at(fSubstates.at(ss1).GetStateIndex()).GetJ();
308  int MInit = 2 * fSubstates.at(ss2).GetM();
309  int Mu = 2 * (fSubstates.at(ss2).GetM() - fSubstates.at(ss1).GetM());
310  int MFinal = 2 * fSubstates.at(ss1).GetM();
311 
312  double tmpZeta = TMath::Power((2*lambda + 1),0.5) * phase * fReaction.ThreeJ(JInit,L2,JFinal,-MInit,Mu,MFinal) * tmpPsi * mePhase;
313 
314  /*std::cout << std::setw(5) << std::left << fSubstates.at(ss1).GetM()
315  << std::setw(5) << std::left << fSubstates.at(ss2).GetM()
316  << std::setw(15) << std::left << tmpZeta

318  << std::setw(15) << std::left << tmpPsi

321  << std::setw(12) << std::left << Power
322  << std::endl;*/

324  // We've got a valid connection between substates for this
325  // multipolarity, so we can add it to the connection
326  tmpConnection.AddLambda(l,tmpXi,tmpPsi,tmpZeta);
327  // We also now have something worth adding to our substate
328  connectionFlag = true;
329  }
330  // If we've created a valid connection, add it to the final
331  // substate:

333  fSubstates.at(ss1).AddConnection(tmpConnection);

444  // steps of the integration we're going to perform. We also redefine everything into

446  //

541  tmpProb.push_back(StateProbabilities[i]);
542  }

551  Q1_Matrix.at(1) = 0.5857864 * dOmega * AmplitudeDerivative.at(1) + 0.1213204 * Q1_Matrix.at(1);

580  StateProbabilities[fSubstates.at(i).GetStateIndex()] = StateProbabilities[fSubstates.at(i).GetStateIndex()] + SubStateProbabilities[i];

619  AmplitudeDerivative = ComputeAmpDerivativeMatrices(Amplitude, Epsilon, Omega);

622  // requires three points:

680

725 // This function evaluates the Amplitude Derivatives at a given point in the integration
726 // process. The amplitude derivative function is defined in the GOSIA manual, Eq. 3.20
727 //****************************************************************************************************//
728 
729 std::vector<TMatrixD> PointCoulEx::ComputeAmpDerivativeMatrices(std::vector<TMatrixD> AmplitudeIn, double Epsilon, double Omega)
730 {
731 
732  // Calculate the exponent from Eq. 3.20 in the GOSIA manual
733  double RAlfa = Epsilon * TMath::SinH(Omega) + Omega;
734 
735  // Clear all vectors and resize them appropriately:
736  std::vector<TMatrixD> AmpDot;
737  AmpDot.resize(2);
738  AmpDot.at(0).ResizeTo(fSubstates.size(),LMax);
739  AmpDot.at(1).ResizeTo(fSubstates.size(),LMax);
740  for(size_t f = 0; f < fSubstates.size(); f++){
741  for(int l = 0; l < LMax; l++){
742  AmpDot.at(0)[f][l] = 0;
743  AmpDot.at(1)[f][l] = 0;
744  }
745  }
746 
747 
748  for(int l=0; l < fNucleus.GetMaxLambda(); l++){
749 
750  // No need to continue with this lambda if there aren't any matrix elements
751  if(fabs(MiscFunctions::GetMaxMatrix(fNucleus.GetMatrixElements().at(l))) < 1e-6)
752  continue;
753 
754  // Grab the collision functions for this lambda, omega, epsilon combination
755  std::vector< std::complex<double> > Q_Matrix = CollisionFunction((l+1),Epsilon,Omega);
756 
757  for(size_t fs = 0; fs < fSubstates.size(); fs++){ // Loop over all substates - this is the "final state" of this step
758  for(size_t c = 0; c < fSubstates.at(fs).GetNconnections(); c++){ // Loop over all connections to this final state
759  Connection tmpConnection = fSubstates.at(fs).GetConnection(c); // Grab the connection
760  int is = tmpConnection.GetConnectedState();
761 
762  // If this connection for this lambda was not set earlier (i.e. didn't
763  // satisfy angular moment conservation or similar) then we don't need
764  // to go any further. This is important where a state is connected by
765  // multiple connections of different multipolarities (e.g. E2/M1)
766  if(!tmpConnection.IsSet(l))
767  continue;
768 
769  int finalState = fSubstates.at(fs).GetStateIndex();
770  int initialState = fSubstates.at(is).GetStateIndex();
771 
772  if(fNucleus.GetMatrixElements().at(l)[finalState][initialState] != 0){ // If there is a non-zero matrix element
773 
774  // Determine the real and imaginary parts of the exponent:
775  double RealEx = TMath::Cos(tmpConnection.GetXi(l) * RAlfa);
776  double ImagEx = TMath::Sin(tmpConnection.GetXi(l) * RAlfa);
777 
778  // Difference in magnetic substates for selection of the
779  // correct collision function:
780  double MuAbs = (int)(fabs(fSubstates.at(is).GetM() - fSubstates.at(fs).GetM()));
781 
782  // Combine all of the variables and the above information
783  // to give real and imaginary amplitudes for this substate
784  // from this connection:
785  double RealRC = (std::real(Q_Matrix.at(MuAbs))*RealEx - std::imag(Q_Matrix.at(MuAbs))*ImagEx) * tmpConnection.GetZeta(l) * fNucleus.GetMatrixElements().at(l)[finalState][initialState];
786  double ImagRC = (std::real(Q_Matrix.at(MuAbs))*ImagEx + std::imag(Q_Matrix.at(MuAbs))*RealEx) * tmpConnection.GetZeta(l) * fNucleus.GetMatrixElements().at(l)[finalState][initialState];
787 
788  // If we're dealing with magnetic transitions we also have
789  // to include an additional element due to the asymmetry in
790  // M-state selection:
791  if(l > 5){
792  if((fSubstates.at(fs).GetM() - fSubstates.at(is).GetM()) < 0){
793  RealRC *= -1;
794  ImagRC *= -1;
795  }
796  }
797 
798  // Add the newly determined amplitude for the final state, multiplied
799  // by the amplitude in the initial state:
800  for(int n=0;n<LMax;n++){
801  double RRC = RealRC * AmplitudeIn.at(0)[is][n] - ImagRC * AmplitudeIn.at(1)[is][n];
802  double IRC = RealRC * AmplitudeIn.at(1)[is][n] + ImagRC * AmplitudeIn.at(0)[is][n];
803  AmpDot.at(0)[fs][n] += IRC;
804  AmpDot.at(1)[fs][n] -= RRC;
805  }
806  }
807  }
808  }
809  }
810 
811 
812  return AmpDot;
813 
814 }

886  output.at(0) = std::complex<double>(Real,Imag);

893  Imag = 0.;
894  output.at(2) = std::complex<double>(Real,Imag);
895 
896  Real = 0.;
897  Imag = ((15. * TMath::Sqrt(5))/16.) * pow(c,3) / pow(b,6);

901

911  output.at(0) = std::complex<double>(Real,Imag);
912

915  output.at(1) = std::complex<double>(Real,Imag);
916

920

922  Imag = ((35 * TMath::Sqrt(35))/16) * a * pow(c,3) / pow(b,8);
923  output.at(3) = std::complex<double>(Real,Imag);
924

927  output.at(4) = std::complex<double>(Real,Imag);
928

930 
931  }

935

938  Imag = 0.;
