176  double CS = 0;
177 
178  TGraph2D *gEvsTheta = InterpolatedEnergyTheta(state,fUseEfficiency);
179 
180  int eSteps = 101;
181  int tSteps = 101;
182  double eMin = fIntegral->GetEnergy(0);
183  double eMax = fIntegral->GetEnergy(nEnergy-1);
184  if(eMin > eMax){
185  double tmp = eMin;
186  eMin = eMax;
187  eMax = tmp;
188  }
189  double tMin = fIntegral->GetCMThetaPoints().at(0).at(0);
190  double tMax = fIntegral->GetCMThetaPoints().at(0).at(fIntegral->GetCMThetaPoints().at(0).size()-1);
191  if(tMin > tMax){
192  double tmp = tMin;
193  tMin = tMax;
194  tMax = tmp;
195  }
196  double tStep = (tMax - tMin) / ((double)tSteps - 1.); // Theta stepsize
197  double eStep = (eMax - eMin) / ((double)eSteps - 1.); // Energy stepsize
198 
199  // We can use Simpson's rule to integrate theta and energy and
200  // for every energy, divide out dE/dX
201  for(int e = 0; e < eSteps; e++){
202  double tmpCS = 0;
203  double energy = eMin + e*eStep;
204  for(int t = 0; t < tSteps; t++){
205  if(t==0 || t==(tSteps-1))
206  tmpCS += gEvsTheta->Interpolate(tMin + t*tStep,energy);
207  else if((t % 2) == 0)
208  tmpCS += gEvsTheta->Interpolate(tMin + t*tStep,energy) * 2;
209  else
210  tmpCS += gEvsTheta->Interpolate(tMin + t*tStep,energy) * 4;
211  }
212  tmpCS *= (tStep / 3.);
213  tmpCS /= fStopping.GetStoppingFit().Eval(energy);
214 
215  if(e == 0 || e == (eSteps-1))
216  CS += tmpCS;
217  else if((e % 2) == 0)
218  CS += tmpCS * 2;
219  else
220  CS += tmpCS * 4;
221  }
222 
223  return CS;
