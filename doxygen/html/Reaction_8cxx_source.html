189 void Reaction::InitReaction(){
190 
191  fM[0] = GetBeamMassMeV();
192  fM[1] = GetTargetMassMeV();
193  fM[2] = fM[0];
194  fM[3] = fM[1];
195 
196  fTLab[0] = GetLabEnergy(); // Kinetic energy of the beam
197  fELab[0] = fTLab[0] + fM[0]; // Total energy of the beam (lab)
198  fPLab[0] = TMath::Sqrt(TMath::Power(fTLab[0],2) + 2 * fTLab[0] * fM[0]); // Momentum of the beam (lab)
199  fVLab[0] = fPLab[0] / fELab[0]; // Velocity of the beam (lab)
200  fGLab[0] = 1 / TMath::Sqrt(1 - TMath::Power(fVLab[0],2)); // Gamma factor of the beam
201 
202  fTLab[1] = 0; // Kinetic energy of the target (0)
203  fELab[1] = fM[1]; // Total energy of the target (lab)
204  fPLab[1] = 0; // Momentum of the target (lab)
205  fVLab[1] = 0; // Velocity of the target (lab)
206  fGLab[1] = 1; // Gamma factor of the target
207 
208  fS = TMath::Power(fM[0],2) + TMath::Power(fM[1],2) + 2 * fELab[0] * fM[1];
209  fInvariantMass = TMath::Sqrt(fS);
210 
211  // CM motion
212 
213  fCmE = fInvariantMass;
214  fCmTi = fCmE - fM[0] - fM[1];
215  fCmTf = fCmTi + fQVal;
216  fCmV = fPLab[0] / (fELab[0] + fM[1]);
217  fCmP = fCmV * fCmE;
218  fCmG = 1 / TMath::Sqrt(1 - TMath::Power(fCmV,2));
219 
220  SetCmFrame(exE);
221

257 double Reaction::ConvertThetaLabToCm(double theta_lab, int part){
258 
259  double theta_cm;
260  if(fGOSIAKin){
261 
262  fAred = 1. + GetBeamMass()/GetTargetMass();
263  fEPmin = GetLabEnergy() - GetExcitationEnergy()*fAred;
264  fTauP = TMath::Sqrt(GetLabEnergy()/fEPmin);
265  fTau = fTauP * GetBeamMass()/GetTargetMass();
266 
267  if(part == 3){
268 
269  double y = TMath::Tan(theta_lab);
270  double t = TMath::Sqrt(TMath::Power(fTauP,2) * TMath::Power(y,4) -
271  (1 + TMath::Power(y,2)) * (TMath::Power(fTauP,2) * TMath::Power(y,2) - 1));
272  double first_solution = (fTauP * TMath::Power(y,2) + t) / (1 + TMath::Power(y,2));
273  first_solution = atan2(TMath::Sqrt(1 - TMath::Power(first_solution,2)),fTau + first_solution);
274  double second_solution = (fTauP * TMath::Power(y,2) - t) / (1 + TMath::Power(y,2));
275  second_solution = atan2(TMath::Sqrt(1 - TMath::Power(second_solution,2)),fTau + second_solution);
276 
277  theta_lab = TMath::Max(first_solution,second_solution);
278 
279  }
280 
281  theta_cm = theta_lab + TMath::ASin(fTau * TMath::Sin(theta_lab));
282  return theta_cm;
283  }
284  else{
285  if(theta_lab > fThetaMax[part])
286  theta_lab = fThetaMax[part];
287 
288  double gtan2 = TMath::Power(TMath::Tan(theta_lab)*fCmG,2);
289  double x = fCmV / fVCm[part];
290  double expr = TMath::Sqrt(1 + gtan2 * (1 - TMath::Power(x,2)));
291 
292  if(TMath::Tan(theta_lab) >= 0)
293  theta_cm = TMath::ACos((-x * gtan2 + expr)/(1 + gtan2));
294  else
295  theta_cm = TMath::ACos((-x * gtan2 - expr)/(1 + gtan2));
296 
297 
298  if(part == 3)
299  theta_cm = TMath::Pi() - theta_cm;
300 
301  return theta_cm;
302  }
