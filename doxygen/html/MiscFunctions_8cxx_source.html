106

108  ja = kp + kpp;
109  jb = 2*k - kp - kpp;
110  double f = TMath::Power(-1,k-kp) * TMath::Power(cosB,ja) * TMath::Power(sinB,jb) * TMath::Sqrt(b1);
111

115  int mas = k - kpp;
116  if(kpp < kp)

120  jc = k - kp - mis;
121  jd = mis;
